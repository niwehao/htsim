// 自动路由算法 — 只需 GPU ID, 无需硬编码路由表 Route* buildRoute(src_gpu, dst_gpu, leaves[], spines[], gpus[]) { src_leaf = src_gpu / 16 // GPU 所在 Leaf dst_leaf = dst_gpu / 16 src_port = src_gpu % 16 // GPU 在 Leaf 上的端口 (0..15) dst_port = dst_gpu % 16 if (src_leaf == dst_leaf) { // 同 Leaf — 1 switch hop Leaf[src_leaf].appendToRoute(route, src_port, dst_port) } else { // 跨 Leaf — 3 switch hops spine = (src_gpu + dst_gpu) % 16 // ECMP 选择 Spine Leaf[src_leaf].appendToRoute(route, src_port, 16 + spine) Spine[spine].appendToRoute(route, src_leaf, dst_leaf) Leaf[dst_leaf].appendToRoute(route, 16 + spine, dst_port) } route += [gpus[dst_gpu].rxQueue, gpus[dst_gpu].appSink] } // 修改 NUM_GPU_NODES 即可自动扩缩: // 256 GPUs → 16 Leaf × 16 Spine = 32 switches // 512 GPUs → 32 Leaf × 16 Spine = 48 switches // 只要 NUM_GPU_NODES / 16 ≤ 32 (Spine 端口数) 即可
